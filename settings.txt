Wallpaper 1:

//your parameter variables go here!
let toeSize = 11 //11 This function operates toe size.
let toeY = 190 //190 This function operates cat length.
let eyeballX = 76 //80 (ranges from 76 to 83 ) This function operates width of the eyes.
let eyeballHeight = 20 //15 This function operates eyeball height.
let bowtieSize = 5 //5 This function operates bowtie Size.
let drawFish =  1 //options: 0, 1, 2, or 3. This function operates fish options.
let eyeColour = true //This function operates left eye colour.   
let chairBG = true //This function operates chair.

function setup_wallpaper(pWallpaper) {
  pWallpaper.output_mode(GRID_WALLPAPER);
  pWallpaper.resolution(NINE_PORTRAIT);
  pWallpaper.show_guide(false); //set this to false when you're ready to print

  //Grid settings
  pWallpaper.grid_settings.cell_width  = 300;
  pWallpaper.grid_settings.cell_height = 300;
  pWallpaper.grid_settings.row_offset  = 50;
}
